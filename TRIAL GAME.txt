<script>
    const foodOptions = [
        { emoji: '🍔', name: 'a hamburger' },
        { emoji: '🍕', name: 'a pizza' },
        { emoji: '🌭', name: 'a hot dog' },
        { emoji: '🍎', name: 'an apple' },
        { emoji: '🌮', name: 'a taco' },
        { emoji: '🍣', name: 'some sushi' }
    ];

    let currentOrder = [];
    let itemsEaten = 0;
    let totalMoney = 0;
    let lastMonster = "";
    let isTransitioning = false;

    // Audio setup
    const sounds = {
        munch: new Audio('munch.mp3'),
        burp: new Audio('burp.mp3'),
        happy: new Audio('monster_happy.mp3'),
        laugh: new Audio('monster_laugh.mp3')
    };

    function stopAllSounds() {
        Object.values(sounds).forEach(s => { s.pause(); s.currentTime = 0; });
    }

    function startGame() {
        document.getElementById('start-screen').classList.add('hidden');
        document.getElementById('main-stage').style.display = 'grid';
        nextCustomer();
    }

    function nextCustomer() {
        stopAllSounds();
        isTransitioning = false;
        itemsEaten = 0;
        
        // Randomize Monster (No repeats)
        let monsters = ['monster_1.gif', 'monster_2.gif', 'monster_3.gif'];
        let choice;
        do {
            choice = monsters[Math.floor(Math.random() * monsters.length)];
        } while (choice === lastMonster);
        lastMonster = choice;
        document.getElementById('monster').src = choice;
        document.getElementById('monster').className = "monster-img";
        document.getElementById('cashier').className = "";

        // UI Reset
        document.getElementById('order-zone').classList.remove('hidden');
        document.getElementById('pouch-zone').classList.add('hidden');
        document.getElementById('result-zone').classList.add('hidden');

        // Create Order
        currentOrder = [...foodOptions].sort(() => 0.5 - Math.random()).slice(0, 3);
        updateSpeech();
        renderMenu();
    }

    function renderMenu() {
        const menu = document.getElementById('menu');
        menu.innerHTML = '';
        foodOptions.forEach(food => {
            const div = document.createElement('div');
            div.className = 'food-item';
            div.innerHTML = food.emoji;
            div.onclick = (e) => feedMonster(food, e);
            menu.appendChild(div);
        });
    }

    function updateSpeech() {
        let text = "I want ";
        currentOrder.forEach((item, index) => {
            if (index < itemsEaten) {
                text += `<span class="highlight-text">${item.name}</span>`;
            } else {
                text += "_______";
            }
            if (index < 2) text += " and ";
        });
        document.getElementById('speech-bubble').innerHTML = text + "!";
    }

    function feedMonster(food, event) {
        if (isTransitioning) return;
        
        const targetItem = currentOrder[itemsEaten];
        if (food.emoji === targetItem.emoji) {
            animateFood(event.target, food.emoji);
            itemsEaten++;
            updateSpeech();
            
            if (itemsEaten === 3) {
                isTransitioning = true;
                setTimeout(showPouchChoice, 1500);
            }
        }
    }

    function animateFood(element, emoji) {
        const rect = element.getBoundingClientRect();
        const monsterRect = document.getElementById('monster').getBoundingClientRect();
        
        const flyer = document.createElement('div');
        flyer.className = 'flying-food';
        flyer.innerHTML = emoji;
        flyer.style.left = rect.left + 'px';
        flyer.style.top = rect.top + 'px';
        document.body.appendChild(flyer);

        sounds.munch.play();

        setTimeout(() => {
            flyer.style.left = (monsterRect.left + 100) + 'px';
            flyer.style.top = (monsterRect.top + 100) + 'px';
            flyer.style.transform = 'scale(2) rotate(360deg)';
        }, 10);

        setTimeout(() => {
            flyer.remove();
            document.getElementById('monster').classList.add('munching');
            setTimeout(() => document.getElementById('monster').classList.remove('munching'), 1200);
        }, 600);
    }

    function showPouchChoice() {
        sounds.burp.play();
        document.getElementById('order-zone').classList.add('hidden');
        document.getElementById('pouch-zone').classList.remove('hidden');
        document.getElementById('speech-bubble').innerText = "BURP! Thank you. Here is a tip...";
    }

    function chooseOutcome(type) {
        let win = false;
        let amount = 0;
        let msg = "";

        if (type === 'safe') {
            amount = 3;
            msg = "You took the safe $3!";
            // Logic: Is there a bigger tip in the pouch?
            const hiddenTip = [5, 7, 10, 0][Math.floor(Math.random() * 4)];
            win = (hiddenTip === 0); // Success if you avoided a potential prank
        } else {
            const outcomes = [5, 7, 10, -1]; // -1 represents poop
            const result = outcomes[Math.floor(Math.random() * outcomes.length)];
            
            if (result === -1) {
                amount = 0;
                msg = "OH NO! Pranked! 💩💩💩";
                win = false;
            } else {
                amount = result;
                msg = `BIG TIP! You got $${amount}!`;
                win = true;
            }
        }

        totalMoney += amount;
        document.getElementById('money-display').innerText = `$${totalMoney}`;
        showFinalFeedback(win, msg);
    }

    function showFinalFeedback(win, msg) {
        document.getElementById('pouch-zone').classList.add('hidden');
        document.getElementById('result-zone').classList.remove('hidden');
        document.getElementById('feedback-text').innerText = msg;

        const monster = document.getElementById('monster');
        const cashier = document.getElementById('cashier');

        if (win) {
            monster.classList.add('bounce');
            cashier.classList.add('bounce');
            sounds.happy.play();
        } else {
            monster.classList.add('shake');
            cashier.classList.add('shake');
            sounds.laugh.play();
        }
    }
</script>
